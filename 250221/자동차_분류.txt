#include <stdio.h>
#include <string.h>

#define MODEL_COUNT 5
#define TYPE_COUNT 4

enum cartype {
    sedan = 1,
    suv = 2,
    truck = 3,
    van = 4
};

const char* car_type(enum cartype type) {
    if (type == sedan) {
        return "sedan";
    } else if (type == suv) {
        return "suv";
    } else if (type == truck) {
        return "truck";
    } else if (type == van) {
        return "van";
    } else {
        return "unknown";
    }
};

struct car_find {
    enum cartype type; // #define TYEP_COUNT 4(SEDAN, SUV, TRUCK, VAN)
    const char *models[MODEL_COUNT]; // #define MODEL_COUNT 5
};

struct car_find carval[TYPE_COUNT] = {
    {sedan, {"sonata", "avante", "k5", "grandeur", "sm6"}},
    {suv, {"tucson", "santaFe", "sorento", "palisade", "qm6"}},
    {truck, {"porter", "bongo", "actros", "mighty", "titan"}},
    {van, {"carnival", "staria", "ray", "sprinter", "vito"}}
};


int main() {
    char model[20];
    
    printf("모델 찾기 : ");
    scanf("%s", model);
    
    for (int i = 0; i < TYPE_COUNT; i++) {
        for (int j = 0; j < MODEL_COUNT; j++) {
            if (strcmp(carval[i].models[j], model) == 0) {
                printf("[%s]는 %s 입니다.\n", model, car_type(carval[i].type));
                break;
            }
        }
    }
}
