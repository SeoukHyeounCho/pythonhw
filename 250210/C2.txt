/*
사용자로부터 한 줄로 정수 세 개를 입력 받아 아래와 같은 연산을 수행하는 프
로그램을 작성하시오.(0으로 나누는 경우에 대한 예외처리 필수)
결과 = (a + b) x c – (a – b) / c
*/

#include <stdio.h>

void main()
{
    
    int a = 0;
    int b = 0;
    int c = 0;
    float d = 0;
   
    printf("숫자를 입력하세요(띄어쓰기로 구분) : ");
    scanf("%d %d %d", &a, &b, &c);
    if(c == 0){
        printf("c의 값은0 이상의 정수를 입력하세요");
    }   else{
        d = (a + b) * c - (a - b) / c;
        printf("%f", d);
    }
}
