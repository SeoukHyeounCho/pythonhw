/*
사용자로부터 시험 점수를 입력받고, 아래 조건에 따라 등급을 부여한 후 추가적인 보너스를 계산하여 출력하는 프로그램을 작성하시오. 

등급 규칙: 90점 이상 → "A 등급" 
80점 이상 89점 이하 → "B 등급" 
70점 이상 79점 이하 → "C 등급" 
60점 이상 69점 이하 → "D 등급" 
60점 미만 → "F 등급" 

보너스 규칙: 
A 등급 → 점수의 10% 추가 보너스 
B 등급 → 점수의 5% 추가 보너스 
C 등급 이하 → 추가 보너스 없음    
*/

#include <stdio.h>

void main()
{
    int a = 0;
    printf("A등급의 경우 점수의 추가 보너스 10%%\n등급의 경우 점수의 추가 보너스 5%%\n");
    scanf("%d", &a);
    
    if(a >= 90){
        a += a * 0.1;
        if (a > 100){
            a = 100;
        }
        printf("A 등급 입니다.\n 보너스 합산 점수 : %d", a);
    } else if(a >= 80 && a <= 89){
        a += a * 0.05;
        printf("B 등급 입니다.\n 보너스 합산점수 : %d", a);
    } else if(a >= 70 && a <= 79){
        printf("C 등급 입니다.\n 점수 : %d", a);
    } else if(a >= 60 && a <= 69){
        printf("D 등급 입니다.\n 점수 : %d", a);
    } else{
        printf("F 등급입니다.\n 점수 : %d", a);
    }
    
}
